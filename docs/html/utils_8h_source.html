25static vector<double> GP_to_dosage( const vector<T> &v, const bool &missingToMean) {
26 vector<double> res( v.size() / 3.0);
27 vector<int> missing;
28
29 // initialize
30 int runningSum = 0, nValid = 0;
31 double value;
32
33 // for each entry in result
34 // use two adjacent values
35 for(int i=0; i<res.size(); i++){
36 // compute dosage from genotype probabilties
37 value = v[3*i]*0 + v[3*i+1]*1 + v[3*i+2]*2;
38
39 // -9 is the missing value, so -18 is diploid
40 if( value == -18){
41 // if missing, set to NaN
42 value = std::numeric_limits<double>::quiet_NaN();
43 missing.push_back(i);
44 }else{
45 // for computing mean
46 runningSum += value;
47 nValid++;
48 }
49
50 // set dosage value
51 res[i] = value;
52 }
53
54 // mean excluding NaNs
55 double mu = runningSum / (double) nValid;
56
57 // if missing values should be set to mean
58 if( missingToMean ){
59 // for each entry with a missing value, set to mean
60 for(const int& i : missing) res[i] = mu;
61 }
62
63 return res;
64}

116static size_t removeDuplicates(vector<T>& vec){
117 unordered_set<T> seen;
118
119 auto newEnd = remove_if(vec.begin(), vec.end(), [&seen](const T& value)
120 {
121 if (seen.find(value) != end(seen))
122 return true;
123
124 seen.insert(value);
125 return false;
126 });
127
128 vec.erase(newEnd, vec.end());
129
130 return vec.size();
131}

156static void standardize( arma::mat &X, const bool &center = true, const bool &scale = true ){
157
158 double sqrt_rdf = sqrt(X.n_rows - 1.0);
159
160 // if center, subtract mean of each column
161 // if scale, divide by sd of each column
162 // Note, norm() does not center the column
163 // this give results consistent with base::scale()
164 // when scale is FALSE
165 for(size_t j=0; j<X.n_cols; j++){
166 if( center ) X.col(j) -= mean(X.col(j));
167 if( scale ) X.col(j) /= norm(X.col(j)) / sqrt_rdf;
168 }
169}

184static void nanToMean( arma::vec & v){
185 // get indeces of finite elements
186 arma::uvec idx = arma::find_finite(v);
187
188 // if number of finite elements is less than the total
189 if( idx.n_elem < v.n_elem ){
190 // compute mean from finite elements
191 double mu = arma::mean( v.elem(idx));
192
193 // replace nan with mu
194 v.replace(arma::datum::nan, mu);
195 }
196}
