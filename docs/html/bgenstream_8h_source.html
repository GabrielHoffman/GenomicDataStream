41static genfile::bgen::View::UniquePtr construct_view(
42 const string & filename,
43 const string & index_filename,
44 const GenomicRanges & gr,
45 const vector<string> & rsids = vector<string>()) {
46
47 if( ! filesystem::exists( filename ) ){
48 throw runtime_error("File does not exist: " + filename);
49 }
50 if( ! filesystem::exists( index_filename ) ){
51 throw runtime_error("File does not exist: " + index_filename);
52 }
53
54 using namespace genfile::bgen ;
55
56 View::UniquePtr view = View::create( filename ) ;
57
58 if( gr.size() > 0){
59 IndexQuery::UniquePtr query = IndexQuery::create( index_filename ) ;
60 for( int i = 0; i < gr.size(); i++ ) {
61 query->include_range( IndexQuery::GenomicRange( gr.get_chrom(i) , gr.get_start(i), gr.get_end(i) ) ) ;
62 }
63 query->include_rsids( rsids ) ;
64 query->initialise() ;
65 view->set_query( query ) ;
66 }
67 return view ;
68}

81 bgenstream(const Param & param) : GenomicDataStream(param) {
82
83 // Initialize genomic regions
84 GenomicRanges gr( param.regions );
85
86 // Initialize view and filter variants
87 view = construct_view( param.file, param.file + ".bgi", gr ) ;
88
89 // number of variants after filtering
90 n_variants_total = view->number_of_variants() ;
91
92 // Filter samples
93 if( param.samples.compare("-") == 0 ){
94 get_all_samples( *view, &number_of_samples, &sampleNames, &requestedSamplesByIndexInDataIndex ) ;
95 }else{
96
97 // get subset of samples
98 vector<string> requestedSamples;
99
100 // boost::erase_all(param.samples, " ");
101 boost::split(requestedSamples, param.samples, boost::is_any_of("\t,\n"));
102
103 get_requested_samples( *view, requestedSamples, &number_of_samples, &sampleNames, &requestedSamplesByIndexInDataIndex ) ;
104 }
105
106 vInfo = new VariantInfo( sampleNames );
107
108 // store probabilities
109 int n = 1e6 * param.initCapacity / (double) (sizeof(double) * number_of_samples * max_entries_per_sample);
110 probs.reserve( n );
111
112 // store dosage
113 n = 1e6 * param.initCapacity / (double) (sizeof(double) * number_of_samples);
114 matDosage.reserve(n);
115 }

224 bool getNextChunk_helper(){
225
226 // clear data, but keep allocated capacity
227 matDosage.clear();
228 vInfo->clear();
229
230 // number of variants in this chunk
231 size_t chunkSize = min(param.chunkSize, n_variants_total - variant_idx_start);
232
233 // if no variants remain, return false
234 if( chunkSize == 0) return false;
235
236 vector<int> data_dimension;
237 data_dimension.push_back(chunkSize);
238 data_dimension.push_back(number_of_samples);
239 data_dimension.push_back(max_entries_per_sample);
240
241 vector<int> ploidy_dimension;
242 ploidy_dimension.push_back( chunkSize ) ;
243 ploidy_dimension.push_back( number_of_samples ) ;
244
245 vector<int> ploidy(ploidy_dimension[0]*ploidy_dimension[1], numeric_limits<int>::quiet_NaN());
246 vector<bool> phased( chunkSize, numeric_limits<bool>::quiet_NaN() ) ;
247
248 string SNPID, rsid, chromosome;
249 genfile::bgen::uint32_t position;
250 vector<string> alleles;
251
252 // Iterate through variants
253 size_t k = 0;
254 for( size_t j = variant_idx_start; j < variant_idx_start + chunkSize; j++ ) {
255
256 // read variant information
257 view->read_variant( &SNPID, &rsid, &chromosome, &position, &alleles ) ;
258
259 // store variant info
260 vInfo->addVariant(chromosome, position, rsid, alleles[0], alleles[1] );
261
262 // read genotype probabilities into DataSetter object
263 DataSetter setter(
264 &ploidy, ploidy_dimension,
265 &probs, data_dimension,
266 &phased,
267 k++,
268 requestedSamplesByIndexInDataIndex
269 );
270
271 view->read_genotype_data_block( setter ) ;
272 }
273 // increament starting position to beginning of next chunk
274 variant_idx_start += chunkSize;
275
276 // Convert to dosage values stored in vector<double>
277 //------------------
278
279 // use probs to create Cube
280 cube C(probs.data(), chunkSize, number_of_samples, max_entries_per_sample, true, true);
281
282 // weight alleles by dosage
283 // With max_entries_per_sample = 4, the unphased coding is
284 // AA/AB/BB/NULL so use weights 0/1/2/0 to conver to dosage
285 // since the last entry doesn't encode valid information
286 // When phased, the coding is [a1 a2] / [a1 a2]
287 // so use weights 0/1/0/1
288 // vec w_unph = {0,1,2,0};
289 // vec w_ph = {0,1,0,1};
290 // compute dosages with weights depend on phasing
291 // w = phased[j] ? w_ph : w_unph;
292 // dsg = C.row_as_mat(j).t() * w;
293 // BUT !!!
294 // in unphased data, they last entry can be NaN
295 // and NaN * 0 is still NaN
296 // so need to drop the last entry _manually_
297
298 vec v, dsg;
299 vec w_unph = {0,1,2};
300 vec w_ph = {0,1,0,1};
301 mat m;
302
303 // compute dosage from Cube
304 // copy results of each variant to vector<double>
305 for(int j=0; j<chunkSize; j++){
306 if( phased[j] ){
307 dsg = C.row_as_mat(j).t() * w_ph;
308 }else{
309 // extract columns 0,1,2
310 // skip 3rd
311 m = C.row_as_mat(j).t();
312 dsg = m.cols(0,2) * w_unph;
313 }
314
315 // replace missing with mean
316 if( param.missingToMean ) nanToMean( dsg );
317
318 // save vector in matDosage
319 memcpy(matDosage.data() + number_of_samples*j, dsg.memptr(), number_of_samples*sizeof(double));
320 }
321
322 return true;
323 }
