63 void initialize(vector<string> regions){
64
65 // if only entry is "." or "",
66 // don't add any regions and return early
67 if( regions.size() == 1 && regions[0].compare(".") == 0) return;
68 if( regions.size() == 1 && regions[0].compare("") == 0) return;
69
70 // Remove duplicate entries, but preserve element order.
71 removeDuplicates( regions );
72
73 vector<string> reg, pos;
74 long p0, p1;
75
76 for(auto const & it : regions){
77
78 // parse
79 boost::split(reg, it, boost::is_any_of(":"));
80 boost::split(pos, reg[1], boost::is_any_of("-"));
81
82 // if only start give, set end to same value
83 if( pos.size() == 1 ) pos.push_back( pos[0] );
84
85 // convert strings to long int
86 p0 = atol(pos[0].c_str());
87 p1 = atol(pos[1].c_str());
88
89 // check region
90 if( (reg.size() != 2) || (pos.size() > 2) || (p1 < p0) || (p0<0) || (p1<0)){
91 throw logic_error("Not a valid genomic range: " + it);
92 }
93 if( !isOnlyDigits(pos[0]) || !isOnlyDigits(pos[1])){
94 throw logic_error("Not a valid genomic range: " + it);
95 }
96
97 // save regions
98 chrom.push_back( reg[0] );
99 start.push_back( p0 );
100 end.push_back( p1 );
101 }
102 }
