40 DelayedStream( Rcpp::RObject robj, const vector<string> &rowNames, const int &chunkSize)
41 : GenomicDataStream(), rowNames(rowNames), chunkSize(chunkSize) {
42
43 if( chunkSize < 1){
44 throw runtime_error("chunkSize must be positive: " + to_string(chunkSize));
45 }
46
47 parsed = new Rtatami::BoundNumericPointer(robj);
48
49 // set current position in matrix to zero

51
52 // set size of intermediate variables
53 const auto& ptr = (*parsed)->ptr;

56
57 if( ptr->nrow() != rowNames.size()){
58 throw runtime_error("DelayedStream: rowNames and nrows must be same size");
59 }
60
61 output.reserve(NC*chunkSize);

177 // original code based on stand-alone beachmat
178 bool getNextChunk_helper(){
179
180 // if end of file reached, return false
181 if( ! continueIterating ) return continueIterating;
182
183 // // get pointer to data
184 const auto& ptr = (*parsed)->ptr;
185
186 // // get workspace as dense row
187 auto wrk = ptr->dense_row();
188
189 // if remaning rows is less than chunkSize,
190 // then set chunkSize to stop at end
191 chunkSize = min(chunkSize, NR - pos);
192
193 // loop through rows
194 for (int i = 0; i < chunkSize; i++) {
195 // get data for row pos + i
196 auto extracted = wrk->fetch(pos + i, buffer.data());
197
198 // copy data into output vector in column i
199 memcpy(output.data() + NC*i, extracted, NC*sizeof(double));
200 }
201
202 // get feature names
203 mInfo->setRowNames(rowNames, pos, pos+chunkSize);
204
205 // increment current position
206 pos += chunkSize;
207
208 // if current position is less than number of rows
209 // return true to continue and get text chunk
210 continueIterating = (pos < NR);
211
212 return true;
213 }
