48 VariantSet( const vector<string> &chrom, const vector<size_t> &position){
49
50 // for each variant
51 // insert (position[i], i) into chromosome hash
52 for(int i=0; i<chrom.size(); i++){
53 map[chrom[i]].push_back( point(position[i], i) );

67 vector<size_t> getIndeces( const string &chrom, const size_t &start, const size_t &end){
68
69 auto vBegin = map[chrom].begin();
70 auto vEnd = map[chrom].end();
71
72 // get iterator to first element in the interval
73 auto it1 = lower_bound(vBegin, vEnd, point(start));
74
75 // get iterator to last element in the interval
76 auto it2 = upper_bound(vBegin, vEnd, point(end));
77
78 vector<size_t> indeces;
79
80 // walk from lower to upper bound
81 // saving index at each step
82 while( it1 != it2){
83 indeces.push_back( it1->index );
84 it1++;
85 }
86
87 return indeces;

92 vector<size_t> getIndeces( const GenomicRanges &gr ){

101 // insert into vector
102 indeces.insert(indeces.end(), idx.begin(), idx.end());
