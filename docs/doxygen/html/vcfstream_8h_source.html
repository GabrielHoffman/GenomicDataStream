42 vcfstream(const Param & param) : GenomicDataStream(param) {
43
44 // check that file exists
45 if( ! filesystem::exists( param.file ) ){
46 throw runtime_error("File does not exist: " + param.file);
47 }
48
49 // check that field was specified
50 if( param.field.compare("") == 0 ){
51 throw runtime_error("Field for VCF/BCF not specified");
52 }

54 // initialize
55 reader = new BcfReader( param.file );
56
57 // Set genomic regions regions
58 setRegions( param.regions );
59
60 reader->setSamples( param.samples );
61
62 // Initialize record with info in header
63 record = new BcfRecord( reader->header );
64
65 // 1: int; 2: float; 3: string; 0: error;
66 fieldType = reader->header.getFormatType( param.field );
67
68 if( fieldType == 3 && param.field.compare("GT"))
69 throw std::runtime_error("field GT is the only supported string type");
70 if( fieldType == 0)
71 throw std::runtime_error("field not found: " + param.field);
72 // initialize varInfo with sample names
73 vInfo = new VariantInfo( reader->SamplesName );
74
75 // Initialize vector with capacity to store nVariants
76 // Note, this allocates memory but does not change .size()
77 // After j variants have been inserted, only entries up to j*nsamples are populated
78 // the rest of the vector is allocated doesn't have valid data
79 int n = 1e6 * param.initCapacity / (double) (sizeof(double) * reader->nsamples);
80 matDosage.reserve( n );
81 }

93 void setRegions(const vector<string> &regions) override {
94
95 validRegions.reserve(regions.size());
96 validRegions.clear();
97
98 // check status of each region
99 // retain only valid, non-empty regions in validRegions
100 for(const string& region : regions){
101
102 switch( reader->getStatus( region ) ){
103 case 1: // region is vaild and not empty
104 validRegions.push_back(region);
105 break;
106
107 case 0: // the region is valid but empty.
108 break;
109
110 case -1: // there is no index file found.
111 throw runtime_error("Could not retrieve index file");
112 break;
113
114 case -2: // the region is not valid
115 throw runtime_error("region was not found: " + region );
116 break;
117 }
118 }
119
120 // initialize to false
121 continueIterating = false;
122
123 // if valid set is not empty
124 if( validRegions.size() > 0 ){
125 // initialize iterator
126 itReg = validRegions.begin();
127
128 reader->setRegion( *itReg );
129
130 continueIterating = true;
131 }
132 }

152 bool getNextChunk( DataChunk<arma::mat> & chunk) override {
153
154 // Update matDosage and vInfo for the chunk
155 bool ret = getNextChunk_helper();
156
157 // keep features with variance >= minVariance
158 // modifies matDosage and vInfo directly
159 applyVarianceFilter(matDosage, vInfo, reader->nsamples, getMinVariance() );
160
161 // mat(ptr_aux_mem, n_rows, n_cols, copy_aux_mem = true, strict = false)
162 bool copy_aux_mem = false; // create read-only matrix without re-allocating memory
163 arma::mat M(matDosage.data(), reader->nsamples, vInfo->size(), copy_aux_mem, true);
164
165 chunk = DataChunk<arma::mat>( M, vInfo );
166
167 return ret;

260 static string variantToString( const BcfRecord &record ) {

265 private:
266 BcfReader *reader = nullptr;
267 BcfRecord *record = nullptr;
268 VariantInfo *vInfo = nullptr;
269 vector<string>::iterator itReg;
270 vector<string> validRegions;
271
272 bool continueIterating;
273 int fieldType;
274
275 // store genotype values
276 // type used based on fieldType
277 // 1: int;
278 // 2: float;
279 // 3: string;
280 // 0: error;
281 vector<int> values_int;
282 vector<float> values_fl;
283
284 vector<double> tmp;
285
286 // stores genotype dosage as doubles, with the next marker inserted at the end
287 // NOTE that when current size is exceeded, .insert() reallocates memory
288 // this can be slow
289 // set using reserve() to set initial capacity so avoid re-alloc
290 vector<double> matDosage;
291
292 bool getNextChunk_helper(){
293
294 // if no valid regions
295 if( validRegions.size() == 0) return false;
296
297 // if end of file reached, return false
298 if( ! continueIterating ) return continueIterating;
299
300 // clear data, but keep allocated capacity
301 matDosage.clear();
302 vInfo->clear();
303
304 // loop thru variant, updating the count each time
305 unsigned int j;
306 for(j=0; j < param.chunkSize; j++){
307
308 // get next variant
309 // if false, reached end of region
310 if( ! reader->getNextVariant( *record ) ){
311
312 // else go to next region
313 itReg++;
314
315 // if this was the last region
316 // set continueIterating so false is retured at next call to
317 // getNextChunk_helper()
318 // then break since no data left
319 if( itReg == validRegions.end()){
320 continueIterating = false;
321 break;
322 }
323
324 // else
325 // initialize the record for this region
326 reader->setRegion( *itReg );
327 reader->getNextVariant( *record );
328 }
329
330 // populate genotype with the values of the current variant
331 // If string, convert to dosage
332 // use values vector based on fieldType
333 switch(fieldType){
334 case 1: // int
335 record->getFORMAT( param.field, values_int);
336 matDosage.insert(matDosage.end(), values_int.begin(), values_int.end());
337 break;
338 case 2: // float
339 record->getFORMAT( param.field, values_fl);
340
341 if( param.field.compare("DS") == 0){
342 // dosage
343 matDosage.insert(matDosage.end(), values_fl.begin(), values_fl.end());
344 }else if( param.field.compare("GP") == 0){
345 // check if site ploidy > 2
346 if( record->ploidy() > 2 ){
347 throw std::runtime_error("GP is not supported for site with ploidy > 2\n " + variantToString(*record));
348 }
349
350 // genotype probabilities
351 vector<double> dsg = GP_to_dosage(values_fl, param.missingToMean);
352
353 matDosage.insert(matDosage.end(), dsg.begin(), dsg.end());
354 }
355 break;
356 case 3: // string. Convert GT to doubles
357
358 // check if site is multi-allelic
359 if( record->isMultiAllelics() ){
360 throw std::runtime_error("GT is not supported for multi-allelic site\n " + variantToString(*record));
361 }
362 // check if site ploidy > 2
363 if( record->ploidy() > 2 ){
364 throw std::runtime_error("GT is not supported for site with ploidy > 2\n " + variantToString(*record));
365 }
366
367 // get GT as int's with vector that is twice as long
368 record->getGenotypes(values_int);
369 tmp = intToDosage( values_int, param.missingToMean );
370 matDosage.insert(matDosage.end(), tmp.begin(), tmp.end());
371 break;
372 }
373
374 // store variant information
375 vInfo->addVariant( record->CHROM(),
376 record->POS(),
377 record->ID(),
378 record->REF(),
379 record->ALT() );
380 }
381
382 bool ret = true;
383
384 // if chunk is empty, return false
385 if( vInfo->size() == 0) ret = false;
386
387 return ret;
388 }
