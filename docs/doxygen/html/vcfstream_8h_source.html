52 vcfstream(const Param & param) : GenomicDataStream(param) {
53
54 // initialize
55 reader = new BcfReader( param.file );
56
57 validRegions.reserve(param.regions.size());
58 validRegions.clear();
59
60 // check status of each region
61 // retain only valid, non-empty regions in validRegions
62 for(const string& region : param.regions){
63
64 switch( reader->getStatus( region ) ){
65 case 1: // region is vaild and not empty
66 validRegions.push_back(region);
67 break;
68
69 case 0: // the region is valid but empty.
70 break;
71
72 case -1: // there is no index file found.
73 throw runtime_error("Could not retrieve index file");
74 break;
75
76 case -2: // the region is not valid
77 throw runtime_error("region was not found: " + region );
78 break;
79 }
80 }
81
82 // initialize iterator
83 itReg = validRegions.begin();
84
85 reader->setRegion( *itReg );
86 reader->setSamples( param.samples );
87
88 // Initialize record with info in header
89 record = new BcfRecord( reader->header );
90
91 // 1: int; 2: float; 3: string; 0: error;
92 fieldType = reader->header.getFormatType(param.field);
93
94 if( fieldType == 3 && param.field.compare("GT"))
95 throw std::runtime_error("field GT is the only supported string type");
96
97 // initialize varInfo with sample names
98 vInfo = new VariantInfo( reader->SamplesName );
99
100 // Initialize vector with capacity to store nVariants
101 // Note, this allocates memory but does not change .size()
102 // After j variants have been inserted, only entries up to j*nsamples are populated
103 // the rest of the vector is allocated doesn't have valid data
104 int n = 1e6 * param.initCapacity / (double) (sizeof(double) * reader->nsamples);
105 matDosage.reserve( n );
106 }

110 ~vcfstream(){
111 if( reader != nullptr) delete reader;
112 if( record != nullptr) delete record;
113 if( vInfo != nullptr) delete vInfo;
114 }

187 static vector<double> intToDosage( const vector<int> &v, const bool &missingToMean) {
188
189 // store and return result
190 vector<double> res( v.size() / 2.0);
191 vector<int> missing;
192
193 // initialize
194 int runningSum = 0, nValid = 0;
195 double value;
196
197 // for each entry in result
198 // use two adjacent values
199 for(int i=0; i<res.size(); i++){
200 value = v[2*i] + v[2*i+1];
201
202 // -9 is the missing value, so -18 is diploid
203 if( value == -18){
204 // if missing, set to NaN
205 value = std::numeric_limits<double>::quiet_NaN();
206 missing.push_back(i);
207 }else{
208 // for computing mean
209 runningSum += value;
210 nValid++;
211 }
212
213 // set dosage value
214 res[i] = value;
215 }
216
217 // mean excluding NaNs
218 double mu = runningSum / (double) nValid;
219
220 // if missing values should be set to mean
221 if( missingToMean ){
222 // for each entry with a missing value, set to mean
223 for(const int& i : missing) res[i] = mu;
224 }
225
226 return res;
227 }

229 private:
230 BcfReader *reader = nullptr;
231 BcfRecord *record = nullptr;
232 VariantInfo *vInfo = nullptr;
233 vector<string>::iterator itReg;
234 vector<string> validRegions;
235
236 bool endOfFile = false;
237 int fieldType;
238
239 // store genotype values
240 // type used based on fieldType
241 // 1: int;
242 // 2: float;
243 // 3: string;
244 // 0: error;
245 vector<int> values_int;
246 vector<float> values_fl;
247
248 vector<double> tmp;
249
250 // stores genotype dosage as doubles, with the next marker inserted at the end
251 // NOTE that when current size is exceeded, .insert() reallocates memory
252 // this can be slow
253 // set using reserve() to set initial capacity so avaoid re-alloc
254 vector<double> matDosage;
255
256 bool getNextChunk_helper(){
257
258 // clear data, but keep allocated capacity
259 matDosage.clear();
260 vInfo->clear();
261
262 // loop thru variant, updating the count each time
263 unsigned int j;
264 for(j=0; j < param.chunkSize; j++){
265
266 // get next variant
267 // if false, reached end of region
268 if( ! reader->getNextVariant( *record ) ){
269
270 // else go to next region
271 itReg++;
272
273 // if this was the last region
274 // set endOfFile and break
275 if( itReg == validRegions.end()){
276 endOfFile = true;
277 break;
278 }
279
280 // else
281 // initialize the record for this region
282 reader->setRegion( *itReg );
283 reader->getNextVariant( *record );
284 }
285
286 // populate genotype with the values of the current variant
287 // If string, convert to dosage
288 // use values vector based on fieldType
289 switch(fieldType){
290 case 1: // int
291 record->getFORMAT( param.field, values_int);
292 matDosage.insert(matDosage.end(), values_int.begin(), values_int.end());
293 break;
294 case 2: // float
295 record->getFORMAT( param.field, values_fl);
296 matDosage.insert(matDosage.end(), values_fl.begin(), values_fl.end());
297 break;
298 case 3: // string. Convert GT to doubles
299
300 // check if site is multi-allelic
301 if( record->isMultiAllelics() ){
302 throw std::runtime_error("GT is not supported for multi-allelic site\n " + variantToString(*record));
303 }
304 // check if site ploidy > 2
305 if( record->ploidy() > 2 ){
306 throw std::runtime_error("GT is not supported for site with ploidy > 2\n " + variantToString(*record));
307 }
308
309 // get GT as int's with vector that is twice as long
310 record->getGenotypes(values_int);
311 tmp = intToDosage( values_int, param.missingToMean );
312 matDosage.insert(matDosage.end(), tmp.begin(), tmp.end());
313 break;
314 }
315
316 // store variant information
317 vInfo->addVariant( record->CHROM(),
318 record->POS(),
319 record->ID(),
320 record->REF(),
321 record->ALT() );
322 }
323
324 // After j variants have been inserted, only entries up to j*nsamples are populated
325 // the rest of the vector is allocated doesn't have valid data.
326 // Therefore, use entires based on .size(), not .capacity()
327 return ! endOfFile;
328 }
