129
130 unique_ptr<GenomicDataStream> gdsStream;
131
132 // Define reader for VCF/VCFGZ/BCF or BGEN
133 // depending on file extension
134 switch( param.fileType ){
135 case VCF:
136 case VCFGZ:
137 case BCF:
138 gdsStream = make_unique<vcfstream>( param );

140 case BGEN:

143 case PGEN:

145#ifndef DISABLE_PLINK

147 break;

149 case OTHER:

151 break;
152 }
