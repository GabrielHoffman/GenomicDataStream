215 bool getNextChunk_helper2 (){
216
217 // clear data, but keep allocated capacity
218 matDosage.clear();
219 vInfo->clear();
220
221 // number of variants in this chunk
222 int chunkSize = min(param.chunkSize, n_requested_variants - currentIdx);
223 chunkSize = max(chunkSize, 0);
224
225 // if no variants remain, return false
226 if( chunkSize == 0) return false;
227
228 // indeces of variants in chunk
229 vector<int> varIdx_sub =
230 {varIdx.begin() + currentIdx,
231 varIdx.begin() + currentIdx + chunkSize};
232
233 // read dosage into matDosage using
234 // 1-based indeces
235 vector<int> varIdx_sub1(varIdx_sub);
236 for(int &i : varIdx_sub1) i++;
237
238 pg->ReadList( matDosage, varIdx_sub1, param.missingToMean);
239
240 // populate vInfo
241 string id, a1, a2,chrom = "";
242 int pos = -1;
243
244 // if PGEN
245 if( genoFileType == PGEN){
246
247 // Get variant info from pvar
248 for(auto i: varIdx_sub){
249 id = pvar->GetVariantId(i);
250 a1 = pvar->GetAlleleCode(i,0);
251 a2 = pvar->GetAlleleCode(i,1);
252
253 // find chrom, pos given id
254 int idx = map_dt_id[id];
255 chrom = dt["CHROM"][idx];
256 pos = atoi(dt["POS"][idx].c_str());
257
258 vInfo->addVariant(chrom, pos, id, a1, a2);
259 }
260 }else{
261 // Get variant info from DataTable from BIM
262 for(auto i: varIdx_sub){
263 chrom = dt["CHROM"][i];
264 pos = atoi(dt["POS"][i].c_str());
265 id = dt["ID"][i];
266 a1 = dt["REF"][i];
267 a2 = dt["ALT"][i];
268
269 vInfo->addVariant(chrom, pos, id, a1, a2);
270 }
271 }
272
273 // increment current index
274 currentIdx += varIdx_sub.size();
275
276 return true;
277 }

281 /* Parse PVAR file of variant positions and IDs
282 */
283 void process_variants(){
284
285 // if file is PGEN
286 if( genoFileType == PGEN ){
287 // Name of .pvar file based on replacing .pgen$
288 fileIdx = regex_replace(param.file, regex("pgen$"), "pvar");
289
290 // Read .pvar file into DataTable
291 // column names are define by line starting with "#CHROM"
292 // lines before this are ignored
293 dt = DataTable( fileIdx, "#CHROM" );
294
295 genoFileType = PGEN;
296
297 // if file is BED
298 }else if( genoFileType == PBED ){
299
300 // Name of .bim file based on replacing .pgen$
301 fileIdx = regex_replace(param.file, regex("bed$"), "bim");
302
303 // Read BIM file with no headerKey
304 dt = DataTable( fileIdx );
305 dt.setColNames({"CHROM", "ID", "CM", "POS", "ALT", "REF"});
306
307 genoFileType = PBED;
308
309 }else{
310 throw logic_error("Not valid genotype file extension: " + param.file);
311 }
312
313 // populate unordered_map linking ID to index
314 // to allow fast search
315 for(int i=0; i<dt["ID"].size(); i++){
316 map_dt_id.emplace(dt["ID"][i], i);
317 }
318
319 // Initialize genomic regions
320 // from delimited string
321 GenomicRanges gr( param.regions );
322
323 // if not empty
324 if( gr.size() != 0){
325 // get indeces of entries in .pvar located
326 // within param.regions
327 // Search is linear time for each interval
328 // varIdx = gr.getWithinIndeces( dt["CHROM"], cast_elements<size_t>(dt["POS"]) );
329
330 // Search is log time for each interval
331 VariantSet vs(dt["CHROM"], cast_elements<size_t>(dt["POS"]));
332 varIdx = vs.getIndeces( gr );
333
334 }else{
335 // else
336 // set entries to seq(0, dt.nrows()-1)
337 varIdx.resize(dt.nrows());
338 iota(begin(varIdx), end(varIdx), 0);
339 }
340
341 // total number of requested variants
342 n_requested_variants = varIdx.size();
343 }

345 void process_samples(){
346
347 DataTable dt2;
348
349 string fileSamples;
350
351 // Get path to samples file
352 // if custom samples file is given
353 if( param.fileSamples.compare("") != 0){
354 fileSamples = param.fileSamples;
355 }else{
356 if( genoFileType == PGEN ){
357 // Name of .psam file based on replacing .pgen$
358 fileSamples = regex_replace(param.file, regex("pgen$"), "psam");
359 }else if( genoFileType == PBED ){
360 // Name of .fam file based on replacing .pgen$
361 fileSamples = regex_replace(param.file, regex("bed$"), "fam");
362 }
363 }
364
365 // Read sample file depending on extension
366 if( regex_search(fileSamples, regex("psam$")) ){
367 // Read .psam file into DataTable
368 // column names are define by line starting with "#IID"
369 // lines before this are ignored
370 dt2 = DataTable(fileSamples, "#IID");
371
372 }else if( regex_search(fileSamples, regex("fam$")) ){
373
374 // Read BIM file with no headerKey
375 dt2 = DataTable( fileSamples );
376
377 // set column names
378 vector<string> names = {"FID", "IID", "PID", "MID", "SEX", "ALT", "PHENO"};
379 vector<string> names_sub(names.begin(), names.begin() + dt.ncols());
380 dt2.setColNames(names_sub);
381 }else{
382 throw logic_error("Not valid sample file extension: " + fileSamples);
383 }
384
385 // Sample names from PSAM file
386 vector<string> SamplesNames = dt2["IID"];
387 n_samples_psam = SamplesNames.size();
388
389 // indeces of samples to include
390 vector<int> sampleIdx;
391
392 // Filter samples
393 // If param.samples contains entries
394 if( param.samples.compare("-") != 0 ){
395
396 // get sample ids from param.samples
397 vector<string> requestedSamples;
398
399 // split delmited string into vector
400 boost::split(requestedSamples, param.samples, boost::is_any_of("\t,\n"));
401
402 // Use unordered_map linking sample id to index
403 // for fast searching
404 // sn = sampleNames
405 unordered_map<string,int> map_sn;
406 for(int i=0; i<SamplesNames.size(); i++){
407 map_sn.emplace(SamplesNames[i], i);
408 }
409
410 // For each requested sample ID,
411 // get its index in the PSAM file
412 for( string & name : requestedSamples){
413 if( map_sn.count(name) == 0){
414 throw logic_error("Sample id not found: " + name);
415 }
416 sampleIdx.push_back( map_sn[name] );
417 }
418 sort(sampleIdx.begin(), sampleIdx.end());
419
420 // Get the requested sample ID's
421 // sorted according to the PSAM file
422 vector<string> requestedSamples_ordered;
423 for(int i : sampleIdx){
424 requestedSamples_ordered.push_back( SamplesNames[i] );
425 }
426
427 // populate VariantInfo with requested sample IDs
428 // in the order from the PSAM file
429 vInfo = new VariantInfo( requestedSamples_ordered );
430 number_of_samples = requestedSamples_ordered.size();
431
432 // convert to 1-based indeces
433 sampleIdx1.assign(sampleIdx.begin(), sampleIdx.end());
434 for(int &i : sampleIdx1) i++;
435 }else{
436 vInfo = new VariantInfo( SamplesNames );
437 number_of_samples = SamplesNames.size();
438 }
439 }
